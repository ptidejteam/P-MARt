56   package org.apache.poi.poifs.dev;
57   
58   import java.io.*;
59   
60   import java.util.*;
61   
62   /**
63    * This class contains methods used to inspect POIFSViewable objects
64    *
65    * @author Marc Johnson (mjohnson at apache dot org)
66    */
67   
68   public class POIFSViewEngine
69   {
70       private static final String _EOL = System.getProperty("line.separator");
71   
72       /**
73        * Inspect an object that may be viewable, and drill down if told
74        * to
75        *
76        * @param viewable the object to be viewed
77        * @param drilldown if true, and the object implements
78        *                  POIFSViewable, inspect the objects' contents
79        *                  (recursively)
80        * @param indentLevel how far in to indent each string
81        * @param indentString string to use for indenting
82        *
83        * @return a List of Strings holding the content
84        */
85   
86       public static List inspectViewable(final Object viewable,
87                                          final boolean drilldown,
88                                          final int indentLevel,
89                                          final String indentString)
90       {
91           List objects = new ArrayList();
92   
93           if (viewable instanceof POIFSViewable)
94           {
95               POIFSViewable inspected = ( POIFSViewable ) viewable;
96   
97               objects.add(indent(indentLevel, indentString,
98                                  inspected.getShortDescription()));
99               if (drilldown)
100              {
101                  if (inspected.preferArray())
102                  {
103                      Object[] data = inspected.getViewableArray();
104  
105                      for (int j = 0; j < data.length; j++)
106                      {
107                          objects.addAll(inspectViewable(data[ j ], drilldown,
108                                                         indentLevel + 1,
109                                                         indentString));
110                      }
111                  }
112                  else
113                  {
114                      Iterator iter = inspected.getViewableIterator();
115  
116                      while (iter.hasNext())
117                      {
118                          objects.addAll(inspectViewable(iter.next(),
119                                                         drilldown,
120                                                         indentLevel + 1,
121                                                         indentString));
122                      }
123                  }
124              }
125          }
126          else
127          {
128              objects.add(indent(indentLevel, indentString,
129                                 viewable.toString()));
130          }
131          return objects;
132      }
133  
134      private static String indent(final int indentLevel,
135                                   final String indentString, final String data)
136      {
137          StringBuffer finalBuffer  = new StringBuffer();
138          StringBuffer indentPrefix = new StringBuffer();
139  
140          for (int j = 0; j < indentLevel; j++)
141          {
142              indentPrefix.append(indentString);
143          }
144          LineNumberReader reader =
145              new LineNumberReader(new StringReader(data));
146  
147          try
148          {
149              String line = reader.readLine();
150  
151              while (line != null)
152              {
153                  finalBuffer.append(indentPrefix).append(line).append(_EOL);
154                  line = reader.readLine();
155              }
156          }
157          catch (IOException e)
158          {
159              finalBuffer.append(indentPrefix).append(e.getMessage())
160                  .append(_EOL);
161          }
162          return finalBuffer.toString();
163      }
164  }   // end public class POIFSViewEngine
