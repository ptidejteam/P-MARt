58   import org.apache.poi.hssf.record.CellValueRecordInterface;
59   import org.apache.poi.hssf.record.Record;
60   import org.apache.poi.hssf.record.UnknownRecord;
61   
62   import java.util.Iterator;
63   import java.util.List;
64   import java.util.TreeMap;

66   /**
67    *
68    * Aggregate value records together.  Things are easier to handle that way.
69    *
70    * @author  andy
71    * @author  Glen Stampoultzis (glens at apache.org)
72    */
73   
74   public class ValueRecordsAggregate
75       extends Record
76   {
77       public final static short sid       = -1000;
78       int                       firstcell = -1;
79       int                       lastcell  = -1;
80       TreeMap                   records   = null;
81       int                       size      = 0;
82   
83       /** Creates a new instance of ValueRecordsAggregate */
84   
85       public ValueRecordsAggregate()
86       {
87           records = new TreeMap();
88       }
89   
90       public void insertCell(CellValueRecordInterface cell)
91       {
92           if (records.get(cell) == null)
93           {
94               size += (( Record ) cell).getRecordSize();
95           }
96           else
97           {
98               size += (( Record ) cell).getRecordSize()
99                       - (( Record ) records.get(cell)).getRecordSize();
100          }
101  
102          // XYLocator xy = new XYLocator(cell.getRow(), cell.getColumn());
103          records.put(cell, cell);
104          if ((cell.getColumn() < firstcell) || (firstcell == -1))
105          {
106              firstcell = cell.getColumn();
107          }
108          if ((cell.getColumn() > lastcell) || (lastcell == -1))
109          {
110              lastcell = cell.getColumn();
111          }
112      }
113  
114      public void removeCell(CellValueRecordInterface cell)
115      {
116          size -= (( Record ) cell).getRecordSize();
117  
118          // XYLocator xy = new XYLocator(cell.getRow(), cell.getColumn());
119          records.remove(cell);
120      }
121  
122      public int getPhysicalNumberOfCells()
123      {
124          return records.size();
125      }
126  
127      public int getFirstCellNum()
128      {
129          return firstcell;
130      }
131  
132      public int getLastCellNum()
133      {
134          return lastcell;
135      }
136  
137      public int construct(int offset, List records)
138      {
139          int k = 0;
140  
141          for (k = offset; k < records.size(); k++)
142          {
143              Record rec = ( Record ) records.get(k);
144  
145              if (!rec.isInValueSection() && !(rec instanceof UnknownRecord))
146              {
147                  break;
148              }
149              if (rec.isValue())
150              {
151                  insertCell(( CellValueRecordInterface ) rec);
152              }
153          }
154          return k;
155      }
156  
157      /**
158       * called by the class that is responsible for writing this sucker.
159       * Subclasses should implement this so that their data is passed back in a
160       * byte array.
161       *
162       * @param offset to begin writing at
163       * @param data byte array containing instance data
164       * @return number of bytes written
165       */
166  
167      public int serialize(int offset, byte [] data)
168      {
169          Iterator itr = records.values().iterator();
170          int      pos = offset;
171  
172          while (itr.hasNext())
173          {
174              pos += (( Record ) itr.next()).serialize(pos, data);
175          }
176          return pos - offset;
177      }
178  
179      /**
180       * called by the constructor, should set class level fields.  Should throw
181       * runtime exception for bad/icomplete data.
182       *
183       * @param data raw data
184       * @param size size of data
185       * @param offset of the record's data (provided a big array of the file)
186       */
187  
188      protected void fillFields(byte [] data, short size, int offset)
189      {
190      }
191  
192      /**
193       * called by constructor, should throw runtime exception in the event of a
194       * record passed with a differing ID.
195       *
196       * @param id alleged id for this record
197       */
198  
199      protected void validateSid(short id)
200      {
201      }
202  
203      /**
204       * return the non static version of the id for this record.
205       */
206  
207      public short getSid()
208      {
209          return sid;
210      }
211  
212      public int getRecordSize()
213      {
214          return size;
215      }
216  
217      public Iterator getIterator()
218      {
219          return records.values().iterator();
220      }
221  }
