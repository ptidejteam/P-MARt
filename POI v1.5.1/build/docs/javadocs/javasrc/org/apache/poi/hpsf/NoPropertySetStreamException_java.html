59   package org.apache.poi.hpsf;
60   
61   /**
62    * <p>This exception is thrown if a format error in a property set
63    * stream is detected or when the input data do not constitute a
64    * property set stream.</p>
65    *
66    * <p>The constructors of this class are analogous to those of its
67    * superclass and documented there.</p>
68    *
69    * @author Rainer Klute (klute@rainer-klute.de)
70    * @version $Id: NoPropertySetStreamException.java,v 1.1 2002/02/14 04:00:59 mjohnson Exp $
71    * @since 2002-02-09
72    */
73   public class NoPropertySetStreamException extends HPSFException
74   {
75   
76       public NoPropertySetStreamException()
77       {
78           super();
79       }
80   
81       public NoPropertySetStreamException(final String msg)
82       {
83           super(msg);
84       }
85   
86       public NoPropertySetStreamException(final Throwable reason)
87       {
88           super(reason);
89       }
90   
91       public NoPropertySetStreamException(final String msg,
92                                           final Throwable reason)
93       {
94           super(msg, reason);
95       }
96   
97   }
