56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   
60   /**
61    * Title:        Gridset Record.<P>
62    * Description:  flag denoting whether the user specified that gridlines are used when
63    *               printing.<P>
64    * REFERENCE:  PG 320 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
65    *
66    * @author Andrew C. Oliver (acoliver at apache dot org)
67    * @author  Glen Stampoultzis (glens at apache.org)
68    *
69    * @version 2.0-pre
70    */
71   
72   public class GridsetRecord
73       extends Record
74   {
75       public final static short sid = 0x82;
76       public short              field_1_gridset_flag;
77   
78       public GridsetRecord()
79       {
80       }
81   
82       /**
83        * Constructs a GridSet record and sets its fields appropriately.
84        *
85        * @param id     id must be 0x82 or an exception will be throw upon validation
86        * @param size  the size of the data area of the record
87        * @param data  data of the record (should not contain sid/len)
88        */
89   
90       public GridsetRecord(short id, short size, byte [] data)
91       {
92           super(id, size, data);
93       }
94   
95       /**
96        * Constructs a GridSet record and sets its fields appropriately.
97        *
98        * @param id     id must be 0x82 or an exception will be throw upon validation
99        * @param size  the size of the data area of the record
100       * @param data  data of the record (should not contain sid/len)
101       * @param offset of the record's data
102       */
103  
104      public GridsetRecord(short id, short size, byte [] data, int offset)
105      {
106          super(id, size, data, offset);
107      }
108  
109      protected void validateSid(short id)
110      {
111          if (id != sid)
112          {
113              throw new RecordFormatException("NOT A Gridset RECORD");
114          }
115      }
116  
117      protected void fillFields(byte [] data, short size, int offset)
118      {
119          field_1_gridset_flag = LittleEndian.getShort(data, 0 + offset);
120      }
121  
122      /**
123       * set whether gridlines are visible when printing
124       *
125       * @param gridset - <b>true</b> if no gridlines are print, <b>false</b> if gridlines are not print.
126       */
127  
128      public void setGridset(boolean gridset)
129      {
130          if (gridset == true)
131          {
132              field_1_gridset_flag = 1;
133          }
134          else
135          {
136              field_1_gridset_flag = 0;
137          }
138      }
139  
140      /**
141       * get whether the gridlines are shown during printing.
142       *
143       * @return gridset - true if gridlines are NOT printed, false if they are.
144       */
145  
146      public boolean getGridset()
147      {
148          return (field_1_gridset_flag == 1);
149      }
150  
151      public String toString()
152      {
153          StringBuffer buffer = new StringBuffer();
154  
155          buffer.append("[GRIDSET]\n");
156          buffer.append("    .gridset        = ").append(getGridset())
157              .append("\n");
158          buffer.append("[/GRIDSET]\n");
159          return buffer.toString();
160      }
161  
162      public int serialize(int offset, byte [] data)
163      {
164          LittleEndian.putShort(data, 0 + offset, sid);
165          LittleEndian.putShort(data, 2 + offset, ( short ) 0x2);
166          LittleEndian.putShort(data, 4 + offset, field_1_gridset_flag);
167          return getRecordSize();
168      }
169  
170      public int getRecordSize()
171      {
172          return 6;
173      }
174  
175      public short getSid()
176      {
177          return this.sid;
178      }
179  }
