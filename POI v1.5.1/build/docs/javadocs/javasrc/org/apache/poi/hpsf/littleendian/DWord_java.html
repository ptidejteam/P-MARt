59   package org.apache.poi.hpsf.littleendian;
60   
61   /**
62    * <p>Represents a double word (4 bytes).</p>
63    *
64    * @author Rainer Klute (klute@rainer-klute.de)
65    * @version $Id: DWord.java,v 1.1 2002/02/14 04:00:59 mjohnson Exp $
66    * @since 2002-02-09
67    */
68   public class DWord extends LittleEndian
69   {
70   
71       /**
72        * <p>Creates a {@link DWord} and reads its value from a byte
73        * array.</p>
74        *
75        * @param src The byte array to read from.
76        *
77        * @param offset The offset of the first byte to read.
78        */
79       public DWord(final byte[] src, final int offset)
80       {
81           super(src, offset);
82       }
83   
84   
85   
86       public final static int LENGTH = 4;
87   
88       public int length()
89       {
90           return LENGTH;
91       }
92   
93   
94   
95       /**
96        * <p>Return the integral value of this {@link DWord}.</p>
97        *
98        * <p><strong>FIXME:</strong> Introduce a superclass for the
99        * numeric types and make this a method of the superclass!</p>
100       */
101      public int intValue()
102      {
103          int value = 0;
104          final int length = length();
105          for (int i = 0; i < length; i++)
106          {
107              final int b = 0xFF & bytes[i];
108              value = value << 8 | b;
109          }
110          return value;
111      }
112  
113  }
