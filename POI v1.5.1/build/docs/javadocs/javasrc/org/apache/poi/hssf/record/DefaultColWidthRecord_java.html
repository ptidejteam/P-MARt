56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   
60   /**
61    * Title:        Default Column Width Record<P>
62    * Description:  Specifies the default width for columns that have no specific
63    *               width set.<P>
64    * REFERENCE:  PG 302 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
65    * @author Andrew C. Oliver (acoliver at apache dot org)
66    * @version 2.0-pre
67    */
68   
69   public class DefaultColWidthRecord
70       extends Record
71   {
72       public final static short sid = 0x55;
73       private short             field_1_col_width;
74   
75       public DefaultColWidthRecord()
76       {
77       }
78   
79       /**
80        * Constructs a DefaultColumnWidth record and sets its fields appropriately.
81        *
82        * @param id     id must be 0x55 or an exception will be throw upon validation
83        * @param size  the size of the data area of the record
84        * @param data  data of the record (should not contain sid/len)
85        */
86   
87       public DefaultColWidthRecord(short id, short size, byte [] data)
88       {
89           super(id, size, data);
90       }
91   
92       /**
93        * Constructs a DefaultColumnWidth record and sets its fields appropriately.
94        *
95        * @param id     id must be 0x55 or an exception will be throw upon validation
96        * @param size  the size of the data area of the record
97        * @param data  data of the record (should not contain sid/len)
98        * @param offset of the record's data
99        */
100  
101      public DefaultColWidthRecord(short id, short size, byte [] data,
102                                   int offset)
103      {
104          super(id, size, data, offset);
105      }
106  
107      protected void validateSid(short id)
108      {
109          if (id != sid)
110          {
111              throw new RecordFormatException("NOT A DefaultColWidth RECORD");
112          }
113      }
114  
115      protected void fillFields(byte [] data, short size, int offset)
116      {
117          field_1_col_width = LittleEndian.getShort(data, 0 + offset);
118      }
119  
120      /**
121       * set the default column width
122       * @param height defaultwidth for columns
123       */
124  
125      public void setColWidth(short height)
126      {
127          field_1_col_width = height;
128      }
129  
130      /**
131       * get the default column width
132       * @return defaultwidth for columns
133       */
134  
135      public short getColWidth()
136      {
137          return field_1_col_width;
138      }
139  
140      public String toString()
141      {
142          StringBuffer buffer = new StringBuffer();
143  
144          buffer.append("[DEFAULTCOLWIDTH]\n");
145          buffer.append("    .colwidth      = ")
146              .append(Integer.toHexString(getColWidth())).append("\n");
147          buffer.append("[/DEFAULTCOLWIDTH]\n");
148          return buffer.toString();
149      }
150  
151      public int serialize(int offset, byte [] data)
152      {
153          LittleEndian.putShort(data, 0 + offset, sid);
154          LittleEndian.putShort(data, 2 + offset, ( short ) 0x2);
155          LittleEndian.putShort(data, 4 + offset, getColWidth());
156          return getRecordSize();
157      }
158  
159      public int getRecordSize()
160      {
161          return 6;
162      }
163  
164      public short getSid()
165      {
166          return this.sid;
167      }
168  }
