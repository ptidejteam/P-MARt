56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   
60   /**
61    * Title:        Protection Revision 4 Record<P>
62    * Description:  describes whether this is a protected shared/tracked workbook<P>
63    *  ( HSSF does not support encryption because we don't feel like going to jail ) <P>
64    * REFERENCE:  PG 373 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
65    * @author Andrew C. Oliver (acoliver at apache dot org)
66    * @version 2.0-pre
67    */
68   
69   public class ProtectionRev4Record
70       extends Record
71   {
72       public final static short sid = 0x1af;
73       private short             field_1_protect;
74   
75       public ProtectionRev4Record()
76       {
77       }
78   
79       /**
80        * Constructs a ProtectionRev4 record and sets its fields appropriately.
81        *
82        * @param id     id must be 0x1af or an exception will be throw upon validation
83        * @param size  the size of the data area of the record
84        * @param data  data of the record (should not contain sid/len)
85        */
86   
87       public ProtectionRev4Record(short id, short size, byte [] data)
88       {
89           super(id, size, data);
90       }
91   
92       /**
93        * Constructs a ProtectionRev4 record and sets its fields appropriately.
94        *
95        * @param id     id must be 0x1af or an exception will be throw upon validation
96        * @param size  the size of the data area of the record
97        * @param data  data of the record (should not contain sid/len)
98        * @param offset of the data
99        */
100  
101      public ProtectionRev4Record(short id, short size, byte [] data,
102                                  int offset)
103      {
104          super(id, size, data, offset);
105      }
106  
107      protected void validateSid(short id)
108      {
109          if (id != sid)
110          {
111              throw new RecordFormatException("NOT A PROTECTION REV 4 RECORD");
112          }
113      }
114  
115      protected void fillFields(byte [] data, short size, int offset)
116      {
117          field_1_protect = LittleEndian.getShort(data, 0 + offset);
118      }
119  
120      /**
121       * set whether the this is protected shared/tracked workbook or not
122       * @param protect  whether to protect the workbook or not
123       */
124  
125      public void setProtect(boolean protect)
126      {
127          if (protect)
128          {
129              field_1_protect = 1;
130          }
131          else
132          {
133              field_1_protect = 0;
134          }
135      }
136  
137      /**
138       * get whether the this is protected shared/tracked workbook or not
139       * @return whether to protect the workbook or not
140       */
141  
142      public short getProtect()
143      {
144          return field_1_protect;
145      }
146  
147      public String toString()
148      {
149          StringBuffer buffer = new StringBuffer();
150  
151          buffer.append("[PROT4REV]\n");
152          buffer.append("    .rowheight      = ")
153              .append(Integer.toHexString(getProtect())).append("\n");
154          buffer.append("[/PROT4REV]\n");
155          return buffer.toString();
156      }
157  
158      public int serialize(int offset, byte [] data)
159      {
160          LittleEndian.putShort(data, 0 + offset, sid);
161          LittleEndian.putShort(data, 2 + offset,
162                                (( short ) 0x02));   // 2 bytes (6 total)
163          LittleEndian.putShort(data, 4 + offset, getProtect());
164          return getRecordSize();
165      }
166  
167      public int getRecordSize()
168      {
169          return 6;
170      }
171  
172      public short getSid()
173      {
174          return this.sid;
175      }
176  }
