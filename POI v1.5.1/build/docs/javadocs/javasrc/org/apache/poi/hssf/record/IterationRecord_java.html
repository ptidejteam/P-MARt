56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   
60   /**
61    * Title:        Iteration Record<P>
62    * Description:  Tells whether to iterate over forumla calculations or not
63    *               (if a formula is dependant upon another formula's result)
64    *               (odd feature for something that can only have 32 elements in
65    *                a formula!)<P>
66    * REFERENCE:  PG 325 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
67    * @author Andrew C. Oliver (acoliver at apache dot org)
68    * @version 2.0-pre
69    */
70   
71   public class IterationRecord
72       extends Record
73   {
74       public final static short sid = 0x11;
75       private short             field_1_iteration;
76   
77       public IterationRecord()
78       {
79       }
80   
81       /**
82        * Constructs an Iteration record and sets its fields appropriately.
83        *
84        * @param id     id must be 0x11 or an exception will be throw upon validation
85        * @param size  the size of the data area of the record
86        * @param data  data of the record (should not contain sid/len)
87        */
88   
89       public IterationRecord(short id, short size, byte [] data)
90       {
91           super(id, size, data);
92       }
93   
94       /**
95        * Constructs an Iteration record and sets its fields appropriately.
96        *
97        * @param id     id must be 0x11 or an exception will be throw upon validation
98        * @param size  the size of the data area of the record
99        * @param data  data of the record (should not contain sid/len)
100       * @param offset of the record's data
101       */
102  
103      public IterationRecord(short id, short size, byte [] data, int offset)
104      {
105          super(id, size, data, offset);
106      }
107  
108      protected void validateSid(short id)
109      {
110          if (id != sid)
111          {
112              throw new RecordFormatException("NOT An ITERATION RECORD");
113          }
114      }
115  
116      protected void fillFields(byte [] data, short size, int offset)
117      {
118          field_1_iteration = LittleEndian.getShort(data, 0 + offset);
119      }
120  
121      /**
122       * set whether or not to iterate for calculations
123       * @param iterate or not
124       */
125  
126      public void setIteration(boolean iterate)
127      {
128          if (iterate)
129          {
130              field_1_iteration = 1;
131          }
132          else
133          {
134              field_1_iteration = 0;
135          }
136      }
137  
138      /**
139       * get whether or not to iterate for calculations
140       *
141       * @return whether iterative calculations are turned off or on
142       */
143  
144      public boolean getIteration()
145      {
146          return (field_1_iteration == 1);
147      }
148  
149      public String toString()
150      {
151          StringBuffer buffer = new StringBuffer();
152  
153          buffer.append("[ITERATION]\n");
154          buffer.append("    .iteration      = ").append(getIteration())
155              .append("\n");
156          buffer.append("[/ITERATION]\n");
157          return buffer.toString();
158      }
159  
160      public int serialize(int offset, byte [] data)
161      {
162          LittleEndian.putShort(data, 0 + offset, sid);
163          LittleEndian.putShort(data, 2 + offset, ( short ) 0x2);
164          LittleEndian.putShort(data, 4 + offset, field_1_iteration);
165          return getRecordSize();
166      }
167  
168      public int getRecordSize()
169      {
170          return 6;
171      }
172  
173      public short getSid()
174      {
175          return this.sid;
176      }
177  }
