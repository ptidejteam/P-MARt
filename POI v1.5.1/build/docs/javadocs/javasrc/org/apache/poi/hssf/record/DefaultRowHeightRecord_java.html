56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   
60   /**
61    * Title:        Default Row Height Record
62    * Description:  Row height for rows with undefined or not explicitly defined
63    *               heights.
64    * REFERENCE:  PG 301 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
65    * @author Andrew C. Oliver (acoliver at apache dot org)
66    * @version 2.0-pre
67    */
68   
69   public class DefaultRowHeightRecord
70       extends Record
71   {
72       public final static short sid = 0x225;
73       private short             field_1_option_flags;
74       private short             field_2_row_height;
75   
76       public DefaultRowHeightRecord()
77       {
78       }
79   
80       /**
81        * Constructs a DefaultRowHeight record and sets its fields appropriately.
82        *
83        * @param id     id must be 0x225 or an exception will be throw upon validation
84        * @param size  the size of the data area of the record
85        * @param data  data of the record (should not contain sid/len)
86        */
87   
88       public DefaultRowHeightRecord(short id, short size, byte [] data)
89       {
90           super(id, size, data);
91       }
92   
93       /**
94        * Constructs a DefaultRowHeight record and sets its fields appropriately.
95        *
96        * @param id     id must be 0x225 or an exception will be throw upon validation
97        * @param size  the size of the data area of the record
98        * @param data  data of the record (should not contain sid/len)
99        * @param offset of the records data
100       */
101  
102      public DefaultRowHeightRecord(short id, short size, byte [] data,
103                                    int offset)
104      {
105          super(id, size, data, offset);
106      }
107  
108      protected void validateSid(short id)
109      {
110          if (id != sid)
111          {
112              throw new RecordFormatException("NOT A DefaultRowHeight RECORD");
113          }
114      }
115  
116      protected void fillFields(byte [] data, short size, int offset)
117      {
118          field_1_option_flags = LittleEndian.getShort(data, 0 + offset);
119          field_2_row_height   = LittleEndian.getShort(data, 2 + offset);
120      }
121  
122      /**
123       * set the (currently unimportant to HSSF) option flags
124       * @param flags the bitmask to set
125       */
126  
127      public void setOptionFlags(short flags)
128      {
129          field_1_option_flags = flags;
130      }
131  
132      /**
133       * set the default row height
134       * @param height    for undefined rows/rows w/undefined height
135       */
136  
137      public void setRowHeight(short height)
138      {
139          field_2_row_height = height;
140      }
141  
142      /**
143       * get the (currently unimportant to HSSF) option flags
144       * @return flags - the current bitmask
145       */
146  
147      public short getOptionFlags()
148      {
149          return field_1_option_flags;
150      }
151  
152      /**
153       * get the default row height
154       * @return rowheight for undefined rows/rows w/undefined height
155       */
156  
157      public short getRowHeight()
158      {
159          return field_2_row_height;
160      }
161  
162      public String toString()
163      {
164          StringBuffer buffer = new StringBuffer();
165  
166          buffer.append("[DEFAULTROWHEIGHT]\n");
167          buffer.append("    .optionflags    = ")
168              .append(Integer.toHexString(getOptionFlags())).append("\n");
169          buffer.append("    .rowheight      = ")
170              .append(Integer.toHexString(getRowHeight())).append("\n");
171          buffer.append("[/DEFAULTROWHEIGHT]\n");
172          return buffer.toString();
173      }
174  
175      public int serialize(int offset, byte [] data)
176      {
177          LittleEndian.putShort(data, 0 + offset, sid);
178          LittleEndian.putShort(data, 2 + offset, ( short ) 0x4);
179          LittleEndian.putShort(data, 4 + offset, getOptionFlags());
180          LittleEndian.putShort(data, 6 + offset, getRowHeight());
181          return getRecordSize();
182      }
183  
184      public int getRecordSize()
185      {
186          return 8;
187      }
188  
189      public short getSid()
190      {
191          return this.sid;
192      }
193  }
