71   public class ValueReferencePtg
72       extends Ptg
73   {
74       private final static int SIZE = 5;
75       public final static byte sid  = 0x44;
76       private short            field_1_row;
77       private short            field_2_col;
78       private BitField         rowRelative = new BitField(0x8000);
79       private BitField         colRelative = new BitField(0x4000);
80   
81       /** Creates new ValueReferencePtg */
82   
83       public ValueReferencePtg()
84       {
85       }
86   
87       /** Creates new ValueReferencePtg */
88   
89       public ValueReferencePtg(byte [] data, int offset)
90       {
91           offset++;   // adjust for ptg
92           field_1_row = LittleEndian.getShort(data, offset + 0);
93           field_2_col = LittleEndian.getShort(data, offset + 2);
94           System.out.println(toString());
95       }
96   
97       public String toString()
98       {
99           StringBuffer buffer = new StringBuffer("[ValueReferencePtg]\n");
100  
101          buffer.append("row = ").append(getRow()).append("\n");
102          buffer.append("col = ").append(getColumnRaw()).append("\n");
103          buffer.append("rowrelative = ").append(isRowRelative()).append("\n");
104          buffer.append("colrelative = ").append(isColRelative()).append("\n");
105          return buffer.toString();
106      }
107  
108      public void writeBytes(byte [] array, int offset)
109      {
110      }
111  
112      public void setRow(short row)
113      {
114          field_1_row = row;
115      }
116  
117      public short getRow()
118      {
119          return field_1_row;
120      }
121  
122      public boolean isRowRelative()
123      {
124          return rowRelative.isSet(field_2_col);
125      }
126  
127      public boolean isColRelative()
128      {
129          return rowRelative.isSet(field_2_col);
130      }
131  
132      public void setColumnRaw(short col)
133      {
134          field_2_col = col;
135      }
136  
137      public short getColumnRaw()
138      {
139          return field_2_col;
140      }
141  
142      public void setColumn(short col)
143      {
144          field_2_col = col;   // fix this
145      }
146  
147      public short getColumn()
148      {
149          return field_2_col;   // fix this
150      }
151  
152      public int getSize()
153      {
154          return SIZE;
155      }
156  
157      public String toFormulaString()
158      {
159          return "NO IDEA YET VALUE REF";
160      }
161  }
