61   package org.apache.poi.hssf.record;
62   
63   import org.apache.poi.util.LittleEndian;
64   
65   /**
66    * Extended SST table info subrecord<P>
67    * contains the elements of "info" in the SST's array field<P>
68    * WE HAVE VERY LITTLE INFORMATION ON HOW TO IMPLEMENT THIS RECORD! (EXTSSST)<P>
69    * @author Andrew C. Oliver (acoliver at apache dot org)
70    * @version 2.0-pre
71    * @see org.apache.poi.hssf.record.ExtSSTRecord
72    */
73   
74   public class ExtSSTInfoSubRecord
75       extends Record
76   {
77       public final static short sid =
78           0xFFF;                                             // only here for conformance, doesn't really have an sid
79       private int               field_1_stream_pos;          // stream pointer to the SST record
80       private short             field_2_bucket_sst_offset;   // don't really understand this yet.
81       private short             field_3_zero;                // must be 0;
82   
83       /** Creates new ExtSSTInfoSubRecord */
84   
85       public ExtSSTInfoSubRecord()
86       {
87       }
88   
89       public ExtSSTInfoSubRecord(short id, short size, byte [] data)
90       {
91           super(id, size, data);
92       }
93   
94       public ExtSSTInfoSubRecord(short id, short size, byte [] data, int offset)
95       {
96           super(id, size, data, offset);
97       }
98   
99       protected void validateSid(short id)
100      {
101  
102          // do nothing
103      }
104  
105      protected void fillFields(byte [] data, short size, int offset)
106      {
107          field_1_stream_pos        = LittleEndian.getInt(data, 0 + offset);
108          field_2_bucket_sst_offset = LittleEndian.getShort(data, 4 + offset);
109          field_3_zero              = LittleEndian.getShort(data, 6 + offset);
110      }
111  
112      public void setStreamPos(int pos)
113      {
114          field_1_stream_pos = pos;
115      }
116  
117      public void setBucketSSTOffset(short offset)
118      {
119          field_2_bucket_sst_offset = offset;
120      }
121  
122      public int getStreamPos()
123      {
124          return field_1_stream_pos;
125      }
126  
127      public short getBucketSSTOffset()
128      {
129          return field_2_bucket_sst_offset;
130      }
131  
132      public String toString()
133      {
134          StringBuffer buffer = new StringBuffer();
135  
136          buffer.append("[EXTSST]\n");
137          buffer.append("    .streampos      = ")
138              .append(Integer.toHexString(getStreamPos())).append("\n");
139          buffer.append("    .bucketsstoffset= ")
140              .append(Integer.toHexString(getBucketSSTOffset())).append("\n");
141          buffer.append("    .zero           = ")
142              .append(Integer.toHexString(field_3_zero)).append("\n");
143          buffer.append("[/EXTSST]\n");
144          return buffer.toString();
145      }
146  
147      public int serialize(int offset, byte [] data)
148      {
149          LittleEndian.putInt(data, 0 + offset, getStreamPos());
150          LittleEndian.putShort(data, 4 + offset, getBucketSSTOffset());
151          LittleEndian.putShort(data, 6 + offset, ( short ) 0);
152          return getRecordSize();
153      }
154  
155      public int getRecordSize()
156      {
157          return 8;
158      }
159  
160      public short getSid()
161      {
162          return this.sid;
163      }
164  }
