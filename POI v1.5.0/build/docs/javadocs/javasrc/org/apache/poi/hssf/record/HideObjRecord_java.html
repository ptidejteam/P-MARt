56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   
60   /**
61    * Title:        Hide Object Record<P>
62    * Description:  flag defines whether to hide placeholders and object<P>
63    * REFERENCE:  PG 321 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
64    * @author Andrew C. Oliver (acoliver at apache dot org)
65    * @version 2.0-pre
66    */
67   
68   public class HideObjRecord
69       extends Record
70   {
71       public final static short sid               = 0x8d;
72       public final static short HIDE_ALL          = 2;
73       public final static short SHOW_PLACEHOLDERS = 1;
74       public final static short SHOW_ALL          = 0;
75       private short             field_1_hide_obj;
76   
77       public HideObjRecord()
78       {
79       }
80   
81       /**
82        * Constructs an HideObj record and sets its fields appropriately.
83        *
84        * @param id     id must be 0x8d or an exception will be throw upon validation
85        * @param size  the size of the data area of the record
86        * @param data  data of the record (should not contain sid/len)
87        */
88   
89       public HideObjRecord(short id, short size, byte [] data)
90       {
91           super(id, size, data);
92       }
93   
94       /**
95        * Constructs an HideObj record and sets its fields appropriately.
96        *
97        * @param id     id must be 0x8d or an exception will be throw upon validation
98        * @param size  the size of the data area of the record
99        * @param data  data of the record (should not contain sid/len)
100       * @param offset of the record's data
101       */
102  
103      public HideObjRecord(short id, short size, byte [] data, int offset)
104      {
105          super(id, size, data, offset);
106      }
107  
108      protected void validateSid(short id)
109      {
110          if (id != sid)
111          {
112              throw new RecordFormatException("NOT A HIDEOBJ RECORD");
113          }
114      }
115  
116      protected void fillFields(byte [] data, short size, int offset)
117      {
118          field_1_hide_obj = LittleEndian.getShort(data, 0 + offset);
119      }
120  
121      /**
122       * set hide object options
123       *
124       * @param hide options
125       * @see #HIDE_ALL
126       * @see #SHOW_PLACEHOLDERS
127       * @see #SHOW_ALL
128       */
129  
130      public void setHideObj(short hide)
131      {
132          field_1_hide_obj = hide;
133      }
134  
135      /**
136       * get hide object options
137       *
138       * @return hide options
139       * @see #HIDE_ALL
140       * @see #SHOW_PLACEHOLDERS
141       * @see #SHOW_ALL
142       */
143  
144      public short getHideObj()
145      {
146          return field_1_hide_obj;
147      }
148  
149      public String toString()
150      {
151          StringBuffer buffer = new StringBuffer();
152  
153          buffer.append("[HIDEOBJ]\n");
154          buffer.append("    .hideobj         = ")
155              .append(Integer.toHexString(getHideObj())).append("\n");
156          buffer.append("[/HIDEOBJ]\n");
157          return buffer.toString();
158      }
159  
160      public int serialize(int offset, byte [] data)
161      {
162          LittleEndian.putShort(data, 0 + offset, sid);
163          LittleEndian.putShort(data, 2 + offset,
164                                (( short ) 0x02));   // 2 bytes (6 total)
165          LittleEndian.putShort(data, 4 + offset, getHideObj());
166          return getRecordSize();
167      }
168  
169      public int getRecordSize()
170      {
171          return 6;
172      }
173  
174      public short getSid()
175      {
176          return this.sid;
177      }
178  }
