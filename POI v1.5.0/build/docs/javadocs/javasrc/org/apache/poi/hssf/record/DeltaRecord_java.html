56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   
60   /**
61    * Title:        Delta Record<P>
62    * Description:  controls the accuracy of the calculations<P>
63    * REFERENCE:  PG 303 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
64    * @author Andrew C. Oliver (acoliver at apache dot org)
65    * @version 2.0-pre
66    */
67   
68   public class DeltaRecord
69       extends Record
70   {
71       public final static short  sid           = 0x10;
72       public final static double DEFAULT_VALUE = 0.0010;   // should be .001
73   
74       // a double is an IEEE 8-byte float...damn IEEE and their goofy standards an
75       // ambiguous numeric identifiers
76       private double             field_1_max_change;
77   
78       public DeltaRecord()
79       {
80       }
81   
82       /**
83        * Constructs a Delta record and sets its fields appropriately.
84        *
85        * @param id     id must be 0x10 or an exception will be throw upon validation
86        * @param size  the size of the data area of the record
87        * @param data  data of the record (should not contain sid/len)
88        */
89   
90       public DeltaRecord(short id, short size, byte [] data)
91       {
92           super(id, size, data);
93       }
94   
95       /**
96        * Constructs a Delta record and sets its fields appropriately.
97        *
98        * @param id     id must be 0x10 or an exception will be throw upon validation
99        * @param size  the size of the data area of the record
100       * @param data  data of the record (should not contain sid/len)
101       * @param offset of record data
102       */
103  
104      public DeltaRecord(short id, short size, byte [] data, int offset)
105      {
106          super(id, size, data, offset);
107      }
108  
109      protected void validateSid(short id)
110      {
111          if (id != sid)
112          {
113              throw new RecordFormatException("NOT A DELTA RECORD");
114          }
115      }
116  
117      protected void fillFields(byte [] data, short size, int offset)
118      {
119          field_1_max_change = LittleEndian.getDouble(data, 0 + offset);
120      }
121  
122      /**
123       * set the maximum change
124       * @param maxChange - maximum rounding error
125       */
126  
127      public void setMaxChange(double maxChange)
128      {
129          field_1_max_change = maxChange;
130      }
131  
132      /**
133       * get the maximum change
134       * @return maxChange - maximum rounding error
135       */
136  
137      public double getMaxChange()
138      {
139          return field_1_max_change;
140      }
141  
142      public String toString()
143      {
144          StringBuffer buffer = new StringBuffer();
145  
146          buffer.append("[DELTA]\n");
147          buffer.append("    .maxchange      = ").append(getMaxChange())
148              .append("\n");
149          buffer.append("[/DELTA]\n");
150          return buffer.toString();
151      }
152  
153      public int serialize(int offset, byte [] data)
154      {
155          LittleEndian.putShort(data, 0 + offset, sid);
156          LittleEndian.putShort(data, 2 + offset, ( short ) 0x8);
157          LittleEndian.putDouble(data, 4 + offset, getMaxChange());
158          return getRecordSize();
159      }
160  
161      public int getRecordSize()
162      {
163          return 12;
164      }
165  
166      public short getSid()
167      {
168          return this.sid;
169      }
170  }
