56   package org.apache.poi.poifs.property;
57   
58   import org.apache.poi.poifs.filesystem.POIFSDocument;
59   
60   /**
61    * Trivial extension of Property for POIFSDocuments
62    *
63    * @author Marc Johnson (mjohnson at apache dot org)
64    */
65   
66   public class DocumentProperty
67       extends Property
68   {
69   
70       // the POIFSDocument this property is associated with
71       private POIFSDocument _document;
72   
73       /**
74        * Constructor
75        *
76        * @param name POIFSDocument name
77        * @param size POIFSDocument size
78        */
79   
80       public DocumentProperty(final String name, final int size)
81       {
82           super();
83           _document = null;
84           setName(name);
85           setSize(size);
86           setNodeColor(_NODE_BLACK);   // simplification
87           setPropertyType(PropertyConstants.DOCUMENT_TYPE);
88       }
89   
90       /**
91        * reader constructor
92        *
93        * @param index index number
94        * @param array byte data
95        * @param offset offset into byte data
96        */
97   
98       protected DocumentProperty(final int index, final byte [] array,
99                                  final int offset)
100      {
101          super(index, array, offset);
102          _document = null;
103      }
104  
105      /**
106       * set the POIFSDocument
107       *
108       * @param doc the associated POIFSDocument
109       */
110  
111      public void setDocument(POIFSDocument doc)
112      {
113          _document = doc;
114      }
115  
116      /**
117       * get the POIFSDocument
118       *
119       * @return the associated document
120       */
121  
122      public POIFSDocument getDocument()
123      {
124          return _document;
125      }
126  
127      /* ********** START extension of Property ********** */
128  
129      /**
130       * give method more visibility
131       *
132       * @return true if this property should use small blocks
133       */
134  
135      public boolean shouldUseSmallBlocks()
136      {
137          return super.shouldUseSmallBlocks();
138      }
139  
140      /**
141       * @return true if a directory type Property
142       */
143  
144      public boolean isDirectory()
145      {
146          return false;
147      }
148  
149      /**
150       * Perform whatever activities need to be performed prior to
151       * writing
152       */
153  
154      protected void preWrite()
155      {
156  
157          // do nothing
158      }
159  
160      /* **********  END  extension of Property ********** */
161  }   // end public class DocumentProperty
