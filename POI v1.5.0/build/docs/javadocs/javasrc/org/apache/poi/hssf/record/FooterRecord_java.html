56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   import org.apache.poi.util.StringUtil;
60   
61   /**
62    * Title:        Footer Record <P>
63    * Description:  Specifies the footer for a sheet<P>
64    * REFERENCE:  PG 317 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
65    * @author Andrew C. Oliver (acoliver at apache dot org)
66    * @author Shawn Laubach (laubach@acm.org) Modified 3/14/02
67    * @version 2.0-pre
68    */
69   
70   public class FooterRecord
71       extends Record
72   {
73       public final static short sid = 0x15;
74       private byte              field_1_footer_len;
75       private String            field_2_footer;
76   
77       public FooterRecord()
78       {
79       }
80   
81       /**
82        * Constructs a FooterRecord record and sets its fields appropriately.
83        *
84        * @param id     id must be 0x15 or an exception will be throw upon validation
85        * @param size  the size of the data area of the record
86        * @param data  data of the record (should not contain sid/len)
87        */
88   
89       public FooterRecord(short id, short size, byte [] data)
90       {
91           super(id, size, data);
92       }
93   
94       /**
95        * Constructs a FooterRecord record and sets its fields appropriately.
96        *
97        * @param id     id must be 0x15 or an exception will be throw upon validation
98        * @param size  the size of the data area of the record
99        * @param data  data of the record (should not contain sid/len)
100       * @param offset of the record's data
101       */
102  
103      public FooterRecord(short id, short size, byte [] data, int offset)
104      {
105          super(id, size, data, offset);
106      }
107  
108      protected void validateSid(short id)
109      {
110          if (id != sid)
111          {
112              throw new RecordFormatException("NOT A FooterRECORD");
113          }
114      }
115  
116      protected void fillFields(byte [] data, short size, int offset)
117      {
118          if (size > 0)
119          {
120              field_1_footer_len = data[ 0 + offset ];
121              field_2_footer     = new String(data, 3 + offset, // [Shawn] Changed 1 to 3 for offset of string
122                                              LittleEndian.ubyteToInt( field_1_footer_len) );
123          }
124      }
125  
126      /**
127       * set the length of the footer string
128       *
129       * @param len  length of the footer string
130       * @see #setFooter(String)
131       */
132  
133      public void setFooterLength(byte len)
134      {
135          field_1_footer_len = len;
136      }
137  
138      /**
139       * set the footer string
140       *
141       * @param footer string to display
142       * @see #setFooterLength(byte)
143       */
144  
145      public void setFooter(String footer)
146      {
147          field_2_footer = footer;
148      }
149  
150      /**
151       * get the length of the footer string
152       *
153       * @return length of the footer string
154       * @see #getFooter()
155       */
156  
157      public short getFooterLength()
158      {
159          return (short)(0xFF & field_1_footer_len); // [Shawn] Fixed needing unsigned byte
160      }
161  
162      /**
163       * get the footer string
164       *
165       * @return footer string to display
166       * @see #getFooterLength()
167       */
168  
169      public String getFooter()
170      {
171          return field_2_footer;
172      }
173  
174      public String toString()
175      {
176          StringBuffer buffer = new StringBuffer();
177  
178          buffer.append("[FOOTER]\n");
179          buffer.append("    .footerlen      = ")
180              .append(Integer.toHexString(getFooterLength())).append("\n");
181          buffer.append("    .footer         = ").append(getFooter())
182              .append("\n");
183          buffer.append("[/FOOTER]\n");
184          return buffer.toString();
185      }
186  
187      public int serialize(int offset, byte [] data)
188      {
189          int len = 4;
190  
191          if (getFooterLength() > 0)
192          {
193              len+=3; // [Shawn] Fixed for two null bytes in the length
194          }
195          LittleEndian.putShort(data, 0 + offset, sid);
196          LittleEndian.putShort(data, 2 + offset,
197                                ( short ) ((len - 4) + getFooterLength()));
198          if (getFooterLength() > 0)
199          {
200              data[ 4 + offset ] = (byte)getFooterLength();
201              StringUtil.putCompressedUnicode(getFooter(), data, 7 + offset); // [Shawn] Place the string in the correct offset
202          }
203          return getRecordSize();
204      }
205  
206      public int getRecordSize()
207      {
208          int retval = 4;
209  
210          if (getFooterLength() > 0)
211          {
212              retval+=3; // [Shawn] Fixed for two null bytes in the length
213          }
214          return retval + getFooterLength();
215      }
216  
217      public short getSid()
218      {
219          return this.sid;
220      }
221  }
