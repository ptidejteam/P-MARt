56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   import org.apache.poi.util.StringUtil;
60   
61   /**
62    * Title:        Write Access Record<P>
63    * Description:  Stores the username of that who owns the spreadsheet generator
64    *               (on unix the user's login, on Windoze its the name you typed when
65    *                you installed the thing)<P>
66    * REFERENCE:  PG 424 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
67    * @author Andrew C. Oliver (acoliver at apache dot org)
68    * @version 2.0-pre
69    */
70   
71   public class WriteAccessRecord
72       extends Record
73   {
74       public final static short sid = 0x5c;
75       private String            field_1_username;
76   
77       public WriteAccessRecord()
78       {
79       }
80   
81       /**
82        * Constructs a WriteAccess record and sets its fields appropriately.
83        *
84        * @param id     id must be 0x5c or an exception will be throw upon validation
85        * @param size  the size of the data area of the record
86        * @param data  data of the record (should not contain sid/len)
87        */
88   
89       public WriteAccessRecord(short id, short size, byte [] data)
90       {
91           super(id, size, data);
92       }
93   
94       /**
95        * Constructs a WriteAccess record and sets its fields appropriately.
96        *
97        * @param id     id must be 0x5c or an exception will be throw upon validation
98        * @param size  the size of the data area of the record
99        * @param data  data of the record (should not contain sid/len)
100       * @param offset of the record data
101       */
102  
103      public WriteAccessRecord(short id, short size, byte [] data, int offset)
104      {
105          super(id, size, data, offset);
106      }
107  
108      protected void validateSid(short id)
109      {
110          if (id != sid)
111          {
112              throw new RecordFormatException("NOT A WRITEACCESS RECORD");
113          }
114      }
115  
116      protected void fillFields(byte [] data, short size, int offset)
117      {
118          field_1_username = new String(data, 3 + offset, data.length - 4);
119      }
120  
121      /**
122       * set the username for the user that created the report.  HSSF uses the logged in user.
123       * @param username of the user who  is logged in (probably "tomcat" or "apache")
124       */
125  
126      public void setUsername(String username)
127      {
128          field_1_username = username;
129      }
130  
131      /**
132       * get the username for the user that created the report.  HSSF uses the logged in user.  On
133       * natively created M$ Excel sheet this would be the name you typed in when you installed it
134       * in most cases.
135       * @return username of the user who  is logged in (probably "tomcat" or "apache")
136       */
137  
138      public String getUsername()
139      {
140          return field_1_username;
141      }
142  
143      public String toString()
144      {
145          StringBuffer buffer = new StringBuffer();
146  
147          buffer.append("[WRITEACCESS]\n");
148          buffer.append("    .name            = ")
149              .append(field_1_username.toString()).append("\n");
150          buffer.append("[/WRITEACCESS]\n");
151          return buffer.toString();
152      }
153  
154      public int serialize(int offset, byte [] data)
155      {
156          String       username = getUsername();
157          StringBuffer temp     = new StringBuffer(0x70 - (0x3));
158  
159          temp.append(username);
160          while (temp.length() < 0x70 - 0x3)
161          {
162              temp.append(
163                  " ");   // (70 = fixed lenght -3 = the overhead bits of unicode string)
164          }
165          username = temp.toString();
166          UnicodeString str = new UnicodeString();
167  
168          str.setString(username);
169          str.setOptionFlags(( byte ) 0x0);
170          str.setCharCount(( short ) 0x4);
171          byte[] stringbytes = str.serialize();
172  
173          LittleEndian.putShort(data, 0 + offset, sid);
174          LittleEndian.putShort(data, 2 + offset,
175                                ( short ) (stringbytes
176                                    .length));   // 112 bytes (115 total)
177          System.arraycopy(stringbytes, 0, data, 4 + offset,
178                           stringbytes.length);
179          return getRecordSize();
180      }
181  
182      public int getRecordSize()
183      {
184          return 116;
185      }
186  
187      public short getSid()
188      {
189          return this.sid;
190      }
191  }
