56   package org.apache.poi.poifs.filesystem;
57   
58   /**
59    * Class DocumentDescriptor
60    *
61    * @author Marc Johnson (mjohnson at apache dot org)
62    * @version %I%, %G%
63    */
64   
65   public class DocumentDescriptor
66   {
67       private POIFSDocumentPath path;
68       private String            name;
69       private int               hashcode = 0;
70   
71       /**
72        * Trivial constructor
73        *
74        * @param path the Document path
75        * @param name the Document name
76        */
77   
78       public DocumentDescriptor(final POIFSDocumentPath path, final String name)
79       {
80           if (path == null)
81           {
82               throw new NullPointerException("path must not be null");
83           }
84           if (name == null)
85           {
86               throw new NullPointerException("name must not be null");
87           }
88           if (name.length() == 0)
89           {
90               throw new IllegalArgumentException("name cannot be empty");
91           }
92           this.path = path;
93           this.name = name;
94       }
95   
96       /**
97        * equality. Two DocumentDescriptor instances are equal if they
98        * have equal paths and names
99        *
100       * @param o the object we're checking equality for
101       *
102       * @return true if the object is equal to this object
103       */
104  
105      public boolean equals(final Object o)
106      {
107          boolean rval = false;
108  
109          if ((o != null) && (o.getClass() == this.getClass()))
110          {
111              if (this == o)
112              {
113                  rval = true;
114              }
115              else
116              {
117                  DocumentDescriptor descriptor = ( DocumentDescriptor ) o;
118  
119                  rval = this.path.equals(descriptor.path)
120                         && this.name.equals(descriptor.name);
121              }
122          }
123          return rval;
124      }
125  
126      /**
127       * calculate and return the hashcode
128       *
129       * @return hashcode
130       */
131  
132      public int hashCode()
133      {
134          if (hashcode == 0)
135          {
136              hashcode = path.hashCode() ^ name.hashCode();
137          }
138          return hashcode;
139      }
140  
141      public String toString()
142      {
143          StringBuffer buffer = new StringBuffer(40 * (path.length() + 1));
144  
145          for (int j = 0; j < path.length(); j++)
146          {
147              buffer.append(path.getComponent(j)).append("/");
148          }
149          buffer.append(name);
150          return buffer.toString();
151      }
152  }   // end public class DocumentDescriptor
