56   package org.apache.poi.poifs.dev;
57   
58   import java.io.*;
59   
60   import java.util.*;
61   
62   import org.apache.poi.poifs.filesystem.POIFSFileSystem;
63   
64   /**
65    * A simple viewer for POIFS files
66    *
67    * @author Marc Johnson (mjohnson at apache dot org)
68    */
69   
70   public class POIFSViewer
71   {
72   
73       /**
74        * Display the contents of multiple POIFS files
75        *
76        * @param args the names of the files to be displayed
77        */
78   
79       public static void main(final String args[])
80       {
81           if (args.length < 0)
82           {
83               System.err.println("Must specify at least one file to view");
84               System.exit(1);
85           }
86           boolean printNames = (args.length > 1);
87   
88           for (int j = 0; j < args.length; j++)
89           {
90               viewFile(args[ j ], printNames);
91           }
92       }
93   
94       private static void viewFile(final String filename,
95                                    final boolean printName)
96       {
97           if (printName)
98           {
99               StringBuffer flowerbox = new StringBuffer();
100  
101              flowerbox.append(".");
102              for (int j = 0; j < filename.length(); j++)
103              {
104                  flowerbox.append("-");
105              }
106              flowerbox.append(".");
107              System.out.println(flowerbox);
108              System.out.println("|" + filename + "|");
109              System.out.println(flowerbox);
110          }
111          try
112          {
113              POIFSViewable fs      =
114                  new POIFSFileSystem(new FileInputStream(filename));
115              List          strings = POIFSViewEngine.inspectViewable(fs, true,
116                                          0, "  ");
117              Iterator      iter    = strings.iterator();
118  
119              while (iter.hasNext())
120              {
121                  System.out.print(iter.next());
122              }
123          }
124          catch (IOException e)
125          {
126              System.out.println(e.getMessage());
127          }
128      }
129  }   // end public class POIFSViewer
