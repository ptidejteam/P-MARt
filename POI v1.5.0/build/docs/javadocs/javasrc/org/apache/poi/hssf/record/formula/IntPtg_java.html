61   package org.apache.poi.hssf.record.formula;
62   
63   import org.apache.poi.util.LittleEndian;
64   
65   /**
66    * Integer (short intger)
67    * Stores a (java) short value in a formula
68    * @author  andy
69    */
70   
71   public class IntPtg
72       extends Ptg
73   {
74       public final static int  SIZE = 3;
75       public final static byte sid  = 0x1e;
76       private short            field_1_value;
77   
78       private String val;
79       private int strlen = 0;
80       /** Creates new IntPtg */
81   
82       public IntPtg()
83       {
84       }
85   
86       public IntPtg(byte [] data, int offset)
87       {
88           setValue(LittleEndian.getShort(data, offset + 1));
89       }
90       
91       protected IntPtg(String formula, int offset) {
92           val = parseString(formula, offset);
93           if (val == null) throw new RuntimeException("WHOOAA there...thats got no int!");
94           strlen=val.length();
95           field_1_value = Short.parseShort(val);
96       }
97       
98   
99       public void setValue(short value)
100      {
101          field_1_value = value;
102      }
103  
104      public short getValue()
105      {
106          return field_1_value;
107      }
108  
109      public void writeBytes(byte [] array, int offset)
110      {
111          array[ offset + 0 ] = sid;
112          LittleEndian.putShort(array, offset + 1, getValue());
113      }
114  
115      public int getSize()
116      {
117          return SIZE;
118      }
119  
120      public String toFormulaString()
121      {
122          return "" + getValue();
123      }
124      
125      private static String parseString(String formula, int pos) {
126          String retval = null;
127          while (pos < formula.length() && Character.isWhitespace(formula.charAt(pos))) {
128              pos++;
129          }
130          
131          if (pos < formula.length()) {
132              if (Character.isDigit(formula.charAt(pos)) ) {
133                  int numpos = pos;
134                  
135                  while (numpos < formula.length() && Character.isDigit(formula.charAt(numpos))){
136                      numpos++;
137                  }
138                  
139                  if (numpos == formula.length() || formula.charAt(numpos) != '.') {
140                      String numberstr = formula.substring(pos,numpos);
141                      try {
142                          int number = Short.parseShort(numberstr);
143                          retval = numberstr;
144                      } catch (NumberFormatException e) {
145                          retval = null;
146                      }
147                  }
148              }
149          }
150          return retval;
151          
152      }
153      
154      public static boolean isNextStringToken(String formula, int pos) {
155          return (parseString(formula,pos) != null);
156      }
157      
158      public int getPrecedence() {
159          return 5;
160      }
161      
162      public int getStringLength() {
163          return strlen;
164      }    
165  }
