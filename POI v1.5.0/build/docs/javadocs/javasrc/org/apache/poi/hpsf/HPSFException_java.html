59   package org.apache.poi.hpsf;
60   
61   /**
62    * <p>This exception is the superclass of all other checked exceptions
63    * thrown in this package. It supports a nested "reason" throwable,
64    * i.e. an exception that caused this one to be thrown.</p>
65    *
66    * @author Rainer Klute (klute@rainer-klute.de)
67    * @version $Id: HPSFException.java,v 1.1 2002/02/14 04:00:59 mjohnson Exp $
68    * @since 2002-02-09
69    */
70   public class HPSFException extends Exception
71   {
72   
73       private Throwable reason;
74   
75   
76   
77       /**
78        * <p>Creates a new {@link HPSFException}.</p>
79        */
80       public HPSFException()
81       {
82           super();
83       }
84   
85   
86   
87       /**
88        * <p>Creates a new {@link HPSFException} with a message
89        * string.</p>
90        */
91       public HPSFException(final String msg)
92       {
93           super(msg);
94       }
95   
96   
97   
98       /**
99        * <p>Creates a new {@link HPSFException} with a reason.</p>
100       */
101      public HPSFException(final Throwable reason)
102      {
103          super();
104          this.reason = reason;
105      }
106  
107  
108  
109      /**
110       * <p>Creates a new {@link HPSFException} with a message string
111       * and a reason.</p>
112       */
113      public HPSFException(final String msg, final Throwable reason)
114      {
115          super(msg);
116          this.reason = reason;
117      }
118  
119  
120  
121      /**
122       * <p>Returns the {@link Throwable} that caused this exception to
123       * be thrown or <code>null</code> if there was no such {@link
124       * Throwable}.</p>
125       */
126      public Throwable getReason()
127      {
128          return reason;
129      }
130  
131  }
