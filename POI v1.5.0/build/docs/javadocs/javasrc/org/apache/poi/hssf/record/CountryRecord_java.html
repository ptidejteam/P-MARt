56   package org.apache.poi.hssf.record;
57   
58   import org.apache.poi.util.LittleEndian;
59   
60   /**
61    * Title:        Country Record (aka WIN.INI country)<P>
62    * Description:  used for localization.  Currently HSSF always sets this to 1
63    * and it seems to work fine even in Germany.  (es geht's auch fuer Deutschland)<P>
64    *
65    * REFERENCE:  PG 298 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
66    * @author Andrew C. Oliver (acoliver at apache dot org)
67    * @version 2.0-pre
68    */
69   
70   public class CountryRecord
71       extends Record
72   {
73       public final static short sid = 0x8c;
74   
75       // 1 for US
76       private short             field_1_default_country;
77       private short             field_2_current_country;
78   
79       public CountryRecord()
80       {
81       }
82   
83       /**
84        * Constructs a CountryRecord and sets its fields appropriately
85        *
86        * @param id     id must be 0x8c or an exception will be throw upon validation
87        * @param size  the size of the data area of the record
88        * @param data  data of the record (should not contain sid/len)
89        */
90   
91       public CountryRecord(short id, short size, byte [] data)
92       {
93           super(id, size, data);
94       }
95   
96       /**
97        * Constructs a CountryRecord and sets its fields appropriately
98        *
99        * @param id     id must be 0x8c or an exception will be throw upon validation
100       * @param size  the size of the data area of the record
101       * @param data  data of the record (should not contain sid/len)
102       * @param offset of the record's data
103       */
104  
105      public CountryRecord(short id, short size, byte [] data, int offset)
106      {
107          super(id, size, data, offset);
108      }
109  
110      protected void validateSid(short id)
111      {
112          if (id != sid)
113          {
114              throw new RecordFormatException("NOT A Country RECORD");
115          }
116      }
117  
118      protected void fillFields(byte [] data, short size, int offset)
119      {
120          field_1_default_country = LittleEndian.getShort(data, 0 + offset);
121          field_2_current_country = LittleEndian.getShort(data, 2 + offset);
122      }
123  
124      /**
125       * sets the default country
126       *
127       * @param country ID to set (1 = US)
128       */
129  
130      public void setDefaultCountry(short country)
131      {
132          field_1_default_country = country;
133      }
134  
135      /**
136       * sets the current country
137       *
138       * @param country ID to set (1 = US)
139       */
140  
141      public void setCurrentCountry(short country)
142      {
143          field_2_current_country = country;
144      }
145  
146      /**
147       * gets the default country
148       *
149       * @return country ID (1 = US)
150       */
151  
152      public short getDefaultCountry()
153      {
154          return field_1_default_country;
155      }
156  
157      /**
158       * gets the current country
159       *
160       * @return country ID (1 = US)
161       */
162  
163      public short getCurrentCountry()
164      {
165          return field_2_current_country;
166      }
167  
168      public String toString()
169      {
170          StringBuffer buffer = new StringBuffer();
171  
172          buffer.append("[COUNTRY]\n");
173          buffer.append("    .defaultcountry  = ")
174              .append(Integer.toHexString(getDefaultCountry())).append("\n");
175          buffer.append("    .currentcountry  = ")
176              .append(Integer.toHexString(getCurrentCountry())).append("\n");
177          buffer.append("[/COUNTRY]\n");
178          return buffer.toString();
179      }
180  
181      public int serialize(int offset, byte [] data)
182      {
183          LittleEndian.putShort(data, 0 + offset, sid);
184          LittleEndian.putShort(data, 2 + offset,
185                                (( short ) 0x04));   // 4 bytes (8 total)
186          LittleEndian.putShort(data, 4 + offset, getDefaultCountry());
187          LittleEndian.putShort(data, 6 + offset, getCurrentCountry());
188          return getRecordSize();
189      }
190  
191      public int getRecordSize()
192      {
193          return 8;
194      }
195  
196      public short getSid()
197      {
198          return this.sid;
199      }
200  }
